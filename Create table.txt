 CREATE TABLE CUSTOMER
 (
 CUSTID CHAR(5) NOT NULL,
 CNAME VARCHAR(20) NOT NULL,
 CCONT CHAR(10) NOT NULL,
 CADD VARCHAR(50) NOT NULL,
 CEMAILID VARCHAR(100) NULL,
 CONSTRAINT PK123 PRIMARY KEY (CUSTID),
 CONSTRAINT U1 UNIQUE(CCONT),
 CONSTRAINT CK1 CHECK(LENGTH(CCONT)=10),
 CONSTRAINT U2 UNIQUE(CEMAILID),
 CONSTRAINT CK2 CHECK(CEMAILID LIKE '%@GMAIL.COM' AND CEMAILID NOT LIKE '@%@GMAIL.COM' )
 )




 CREATE TABLE PRODUCT 
 (
 PID CHAR(5) PRIMARY KEY,
 PNAME VARCHAR(50) NOT NULL,
 PCOST VARCHAR(50) NOT NULL,
 PCATEGORY VARCHAR(50) NOT NULL,
 PBRAND VARCHAR(50) NOT NULL,
 PEXPIRY DATE);


CREATE TABLE ORDERS
(
 CUSTID CHAR(5) NOT NULL,
 PID CHAR(5) PRIMARY KEY,
ORDERSTATUS VARCHAR(10) NOT NULL,
CONSTRAINT FK1 FOREIGN KEY(CUSTID) REFERENCES CUSTOMER(CUSTID),
CONSTRAINT FK2 FOREIGN KEY(CUSTID) REFERENCES PRODUCT(PID)
);
